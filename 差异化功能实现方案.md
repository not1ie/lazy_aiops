```python
# 时间序列预测模型
from prophet import Prophet
import pandas as pd

def predict_resource_usage(metric_data):
    """
    预测资源使用趋势
    """
    df = pd.DataFrame({
        'ds': metric_data['timestamps'],
        'y': metric_data['values']
    })
    
    model = Prophet(
        changepoint_prior_scale=0.05,
        seasonality_mode='multiplicative'
    )
    model.fit(df)
    
    # 预测未来2小时
    future = model.make_future_dataframe(periods=24, freq='5min')
    forecast = model.predict(future)
    
    # 检测异常
    threshold = df['y'].mean() + 2 * df['y'].std()
    alerts = forecast[forecast['yhat'] > threshold]
    
    return {
        'forecast': forecast,
        'alerts': alerts,
        'threshold': threshold
    }
```

### 1.3 自动化建议引擎

#### 规则引擎
```go
// 建议规则
type SuggestionRule struct {
    ID          string
    Name        string
    Condition   func(*Metrics) bool
    Suggestion  string
    Priority    int
    AutoFix     func() error
}

var rules = []SuggestionRule{
    {
        ID:   "high_cpu",
        Name: "CPU使用率过高",
        Condition: func(m *Metrics) bool {
            return m.CPU > 80 && m.Duration > 5*time.Minute
        },
        Suggestion: "建议：1. 检查进程占用 2. 优化代码 3. 扩容服务器",
        Priority:   1,
        AutoFix: func() error {
            // 自动扩容逻辑
            return scaleUp()
        },
    },
    {
        ID:   "slow_query",
        Name: "数据库查询慢",
        Condition: func(m *Metrics) bool {
            return m.QueryTime > 500*time.Millisecond
        },
        Suggestion: "建议：1. 添加索引 2. 优化SQL 3. 增加缓存",
        Priority:   2,
    },
}

// 分析并生成建议
func AnalyzeAndSuggest(metrics *Metrics) []Suggestion {
    var suggestions []Suggestion
    
    for _, rule := range rules {
        if rule.Condition(metrics) {
            suggestions = append(suggestions, Suggestion{
                Rule:       rule,
                Timestamp:  time.Now(),
                CanAutoFix: rule.AutoFix != nil,
            })
        }
    }
    
    // 按优先级排序
    sort.Slice(suggestions, func(i, j int) bool {
        return suggestions[i].Rule.Priority < suggestions[j].Rule.Priority
    })
    
    return suggestions
}
```

```javascript
function updateNodeStatus(nodeId, status) {
    const node = document.querySelector(`[data-node-id="${nodeId}"]`);

    node.classList.add(status);
    
    // 添加动画效果
    if (status === 'running') {
        node.classList.add('pulse');
    }
}
```

### 3.2 实现示例
```javascript
import * as THREE from 'three';
import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls';

class ServiceTopology {
    constructor(container) {
        this.scene = new THREE.Scene();
        this.camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        this.renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        
        this.init(container);
    }
    
    init(container) {
        // 设置渲染器
        this.renderer.setSize(container.clientWidth, container.clientHeight);
        container.appendChild(this.renderer.domElement);
        
        // 设置相机
        this.camera.position.z = 50;
        
        // 添加控制器
        this.controls = new OrbitControls(this.camera, this.renderer.domElement);
        this.controls.enableDamping = true;
        
        // 添加光源
        const light = new THREE.AmbientLight(0xffffff, 0.5);
        this.scene.add(light);
        
        const directionalLight = new THREE.DirectionalLight(0xffffff, 0.5);
        directionalLight.position.set(10, 10, 10);
        this.scene.add(directionalLight);
        
        // 开始渲染
        this.animate();
    }
    
    // 添加服务节点
    addServiceNode(service) {
        const geometry = new THREE.SphereGeometry(2, 32, 32);
        const material = new THREE.MeshPhongMaterial({
            color: this.getStatusColor(service.status),
            emissive: 0x072534,
            shininess: 100
        });
        
        const sphere = new THREE.Mesh(geometry, material);
        sphere.position.set(service.x, service.y, service.z);
        sphere.userData = service;
        
        this.scene.add(sphere);
        
        // 添加标签
        this.addLabel(service.name, sphere.position);
        
        return sphere;
    }
    
    // 添加连接线
    addConnection(from, to, traffic) {
        const points = [from.position, to.position];
        const geometry = new THREE.BufferGeometry().setFromPoints(points);
        const material = new THREE.LineBasicMaterial({
            color: this.getTrafficColor(traffic),
            linewidth: traffic / 100
        });
        
        const line = new THREE.Line(geometry, material);
        this.scene.add(line);
        
        // 添加流量动画
        this.animateTraffic(line, from, to);
    }
    
    // 动画循环
    animate() {
        requestAnimationFrame(() => this.animate());
        this.controls.update();
        this.renderer.render(this.scene, this.camera);
    }
    
    // 根据状态获取颜色
    getStatusColor(status) {
        const colors = {
            healthy: 0x30d158,
            warning: 0xff9f0a,
            error: 0xff453a,
            unknown: 0x98989d
        };
        return colors[status] || colors.unknown;
    }
}
```

### 6.1 通知聚合
```javascript
class NotificationAggregator {
    constructor() {
        this.notifications = [];
        this.rules = [
            {
                // 相同服务的告警聚合
                match: (n1, n2) => n1.service === n2.service,
                aggregate: (notifications) => ({
                    title: `${notifications[0].service} 服务异常`,
                    message: `共 ${notifications.length} 条告警`,
                    details: notifications
                })
            }
        ];
    }
    
    add(notification) {
        this.notifications.push(notification);
        this.aggregate();
    }
    
    aggregate() {
        // 应用聚合规则
        const aggregated = [];
        const processed = new Set();
        
        for (const notification of this.notifications) {
            if (processed.has(notification.id)) continue;
            
            const related = this.findRelated(notification);
            if (related.length > 1) {
                aggregated.push(this.mergeNotifications(related));
                related.forEach(n => processed.add(n.id));
            } else {
                aggregated.push(notification);
                processed.add(notification.id);
            }
        }
        
        return aggregated;
    }
}
```
